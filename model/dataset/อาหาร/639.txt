อเมริกาส์เน็กซต์ท็อปโมเดล ฤดูกาลที่ 3 ออกอากาศเมื่อวันที่ 22 กันยายน พ.ศ. 2547 เป็นฤดูกาลแรกที่ได้รับการสนับสนุนจากเครื่องสำอาง CoverGirl และเพิ่มจำนวนผู้เข้าแข่งขันเป็น 14 คน โดยคำโฆษณาประจำฤดูนี้ก็คือ "Beauty In Progress" และเป็นหนึ่งในฤดูกาลที่มียอดผู้ชมสูงเป็นประวัติการณ์ในรอบ 6 คนสุดท้ายของฤดูกาลนี้ ได้ไปแข่งขันต่อที่ กรุงโตเกียว ประเทศญี่ปุ่น  โดยผู้ชนะจะได้เซ็นสัญญากับบริษัทโมเดลลิ่งชั้นนำ   Ford Models ได้ถ่ายแบบกับนิตยสาร ELLE และเซ็นสัญญา 100,000 เหรียญ กับเครื่องสำอาง CoverGirl และผู้ชนะก็คือ อีวา พิกฟอร์ด วัย 20 ปีจาก ลอสแอนเจลิส  แคลิฟอร์เนีย ซึ่งเป็นผู้ชนะผิวสีคนแรกที่มีเชื้อสายแอฟริกัน อเมริกันออกอากาศ : 22 กันยายน พ.ศ. 2547ผู้เข้าแข่งขันทั้ง 34 คนที่ผ่านเข้าถึงรอบสุดท้ายก่อนรอบคัดเลือก ได้เดินทางมาที่ ลอสแอนเจลิส แคลิฟอร์เนีย  สาวๆ จัดงานปาร์ตี้ริมสระเพื่อฉลองที่ผ่านการคัดเลือกกันอย่างสนุกสนาน ในขณะที่สาวๆ บางกลุ่มออกไปเที่ยวคลับ ทิฟฟานี่ ริชาร์ดสันได้เกิดมีเรื่องทะเลาะกับสาวในบาร์ ซึ่งนั่นทำให้เธอไม่ผ่านรอบคัดเลือก 20 คนผู้เข้าแข่งขันต้องผ่านการสัมภาษณ์แบบตัวต่อตัวกับกรรมการ หญิงสาว 14 คนต้องถูกคัดออกและเหลือสาวเพียง 20 คนที่จะผ่านการคัดเลือกในรอบสุดท้าย  และก่อนที่จะคัดเลือกรอบสุดท้าย อแมนด้า ได้เปิดเผยความจริงให้ทุกคนรู้เกี่ยวกับตาของเธอว่า เธอเป็นโรคแก้วตาเปลี่ยนสีและภายในอีกไม่กี่ปี ตาของเธอจะบอดสนิทซึ่งทำให้ผู้เข้าแข่งขันคนอื่นถึงกับต้องหลั่งน้ำตาจากนั้นการคัดเลือกรอบสุดท้ายก็เริ่มขึ้น ซึ่งมีสาวๆ ที่ผ่านการคัดเลือกทั้งหมด 14 คน และจากนั้นสาวๆ ที่ผ่านการคัดเลือกทั้งหมดก็เดินทางเข้าสู่นิวยอร์กเพื่อทำการแข่งขันกันต่อไปออกอากาศ : 29 กันยายน พ.ศ. 2547ผู้เข้าแข่งขันทั้ง 14 คน ถูกส่งมาที่อ่าวมอนเทโก้ จาไมก้า เพื่อถ่ายรูปครั้งแรกในชุดว่ายน้ำโดยมีชายฝั่งจาไมก้าเป็นฉากด้านหลัง ก่อนที่จะถ่ายรูป เจย์ แมนูเอล ได้บอกกับผู้เข้าแข่งขันว่า มีที่นอนสำหรับสาวๆ เพียง 13 ที่เท่านั้น และจะมีคนนึงที่จะต้องตกรอบและเดินทางกลับบ้านในวันต่อมาในขณะที่กำลังทานอาหารกลางวันกันอยู่ สาวๆ ก็ได้รับจดหมายที่เขียนชื่อผู้ที่ต้องตกรอบเป็นคนแรกเอาไว้ และคนที่ต้องกลับบ้านเป็นคนแรกก็คือ แม็กเดลิน่าผู้ที่ถูกคัดออกนอกห้องตัดสิน : แม็กเดลิน่า รีวาสหลังจากที่กลับมาถึงนิวยอร์ก สาวๆ ก็ได้ย้ายเข้าห้องพักที่โรงแรม วอลดอร์ฟ แอสโทเรีย หลังจากที่แบ่งห้องกันแล้ว อีวาไม่ได้อยู่ห้องเดียวกับแอนอย่างที่ตั้งใจไว้ตอนแรก ซึ่งเธอก็ดูไม่ค่อยสนใจอะไรเท่าไหร่นัก ทำให้แอนเสียใจมากโดยเธอพูดออกมาว่าเธอรักอีวามากและอยากให้อีวายืนอยู่เคียงข้างกับเธอในงานแต่งงาน   หลังจากนั้นไทร่าได้เข้ามาหาผู้เข้าแข่งขันเพื่อทำความรู้จักกับทุกๆ คนให้มากขึ้น เจนนิเฟอร์ ยาย่า และทอคคาร่า ได้บอกกับไทร่าว่าพวกเธอรู้สึกรำคาญเคลลี่ รู้สึกว่าเธอเหมือนเด็กเหลือขอถูกตามใจ ซึ่งทำให้เคลลี่เสียใจเป็นอย่างมาก แต่ก็ได้อแมนด้ามาช่วยปลอบใจทำให้เธอรู้สึกดีขึ้นและรู้สึกว่าอย่างน้อยก็มีคนนึงที่ยังเป็นเพื่อนกับเธอ ในการคัดออกครั้งแรก ภาพของลีอาห์และแอนทำให้พวกเธอต้องกลายเป็นสองคนสุดท้าย และลีอาห์เป็นต้องถูกคัดออกเพราะกรรมการรู้สึกว่าเธอผิดจังหวะตลอดและไร้ทักษะในการเป็นนางแบบมากจนเกินไปผู้ที่ถูกเรียกชื่อคนแรก : ยาย่า ดาคอสต้าผู้ที่ยืนเป็นสองคนสุดท้าย : แอน มาร์คลีย์ และ ลีอาห์ ดาร์โรว์ผู้ที่ถูกคัดออก : ลีอาห์ ดาร์โรว์ช่างถ่ายภาพ : นิค คาร์ดัลลิซิโอแขกรับเชิญพิเศษ : เจ.อเล็กซานเดอร์, มิเชลล์ ม็อค, ดัคลาส บาร์เซลลอส, แมธูว แอนเดอร์สัน, แดนิโลสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : อีวา พิกฟอร์ดออกอากาศ : 6 ตุลาคม พ.ศ. 2547ถึงเวลาที่สาวๆ จะต้องถูกจับแปลงโฉมเปลี่ยนลุคส์ใหม่  ผมของเจนนิเฟอร์ถูดตัดออกให้สั้นลง และถูกนำใปบริจาคให้กับมูลนิธิ ล็อก ออฟ เลิฟ  ส่วนอแมนด้าถูกเปลี่ยนสีผมจาก สีบรูเน็ต เป็นสีไอซ์ซี่บลอนด์ในการแข่งขันสัปดาห์นี้ สาวๆต้องแข่งกันแต่งหน้าให้ออกมาเป็นอย่างเป็นธรรมชาติ โนเรลล์สามารถแต่งหน้าออกมาได้ดีที่สุด เธอจึงเป็นผู้ชนะไป และเธอได้เลือก อีวา แอน และคริสตี้ ให้ไปงานปาร์ตี้กับเธอด้วย ส่วนสาวๆคนอื่นต้องไปเป็นเด็กเสิร์ฟในงาน ตลอดทั้งสัปดาห์นี้ ผู้เข้าแข่งขันเริ่มรู้สึกถึงพฤติกรรมการกินที่ผิดปกติของแคสซี่ที่ไม่ยอมกินอะไรเลย และในระหว่างถ่ายภาพซึ่งสัปดาห์นี้สาวๆ จะต้องถ่ายภาพโฆษณาให้กับนีเวีย โดยที่ต้องไม่แต่งหน้าเลยแม้แต่นิดเดียว  ในระหว่างถ่ายภาพ แคสซี่ได้บอกกับอแมนด้าว่าเธออาจเป็นโรคบูลิมิค เมื่อสาวๆ คนอื่นรู้เรื่องเข้าก็รู้สึกเป็นห่วงเธอมากที่ห้องตัดสิน ผู้เข้าแข่งขันได้ดูภาพก่อนและหลังรีทัชของตัวเอง  กรรมการประทับใจรูปของอแมนด้าที่ดูไม่ต่างกันเลยไม่ว่าก่อนหรือหลังรีทัช ภาพของยาย่า อีวา นิโคล และทอคคาร่า ได้รับคำชมเชยมาก และกรรมการรู้สึกว่าภาพของแคสซี่ดูน่าเบื่อ เคลลี่และจูลี่ตกเป็นสองคนสุดท้าย เพราะเคลลี่โทษว่าที่ภาพของเธอออกมาดูไม่มีพลัง เพราะว่าคณะกรรมการไม่เลือกรูปอื่นที่ดีกว่านี้ และคณะกรรมการผิดหวังมากที่จูลี่บอกว่าเธอไม่ได้อยากเป็นนางแบบแต่เธออยากใช้มันเรียนรู้เพื่อปูทางสำหรับเปิดบริษัทเสื้อผ้า และจูลี่ต้องถูกคัดออกเพราะกรรมการรู้สึกว่าเธอไม่ได้อยากเป็นนางแบบและยังคิดใช้อาชีพนี้สำหรับปูทางไปสู่อาชีพอื่นอีกด้วยผู้ที่ถูกเรียกชื่อคนแรก : อแมนด้า สวาฟฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : จูลี่ ไทตัส และ เคลลี่ จาค็อบผู้ที่ถูกคัดออก : จูลี่ ไทตัสช่างถ่ายภาพ : ทรอย วาร์ดแขกรับเชิญพิเศษ : เคียร่า คาบูคูรุ, แอนโทนี่ ปาเลอโม, เลโอนาร์ด ซากามิ, เดบร้า เจลิแมน เอ็ม ดี, ดร.ซาราบิ, แดนิโลสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ยาย่า ดาคอสต้าออกอากาศ : 13 ตุลาคม พ.ศ. 2547ในสัปดาห์นี้ สาวๆที่เหลือต้องมาหัดเดินแบบกับมิสเจ และต่อมาสาวๆ ก็ต้องประหลาดใจเมื่อรู้ว่าจะต้องเดินแบบให้กับ เฮทเทอเรท สำหรับโนเรลล์ เธอรู้สึกเหมือนฝันเป็นจริงเพราะเธอชื่นชอบ เฮทเทอเรท มานาน อแมนด้ารู้สึกกังวลว่าโรคแก้วตาเปลี่ยนสีจะทำให้เธอถูกคัดออกเพราะเธอจะมองไม่เห็นอะไรเลยในเวลากลางคืน  และเมื่อถึงเวลาเดินแบบ โนเรลล์สะดุดล้มบนรันเวย์  ท่าเดินของอีวาโดดเด่นจนเหมือนว่าโชว์ทั้งหมดเป็นของเธอคนเดียว และทอคคาร่าคิดว่าผู้ชมปรบมือให้เธอมากที่สุดเพราะความเซ็กซี่ของเธอ แต่อีวาคิดว่าเป็นเพราะหน้าอกของทอคคาร่าทะลักออกมามากกว่า  อแมนด้าทำพลาดไปเล่าเรื่องตาของเธอให้มิสเจฟัง ทำให้ผู้เข้าแข่งขันคนอื่นพากันคิดว่าอแมนด้าพูดโกหกเพื่อที่จะได้การดูแลเป็นพิเศษกว่าคนอื่น แต่สุดท้ายอีวาเป็นผู้ชนะการแข่งขันและรางวัลของเธอก็คือได้ขึ้นเรือยอร์ชสุดหรูกับโยแอนนา ผู้ชนะจากฤดูกาลที่สอง และเธอเลือกให้เคลลี่กับแอนให้ไปกับเธอด้วยคริสตัลของอแมนด้าหายไปและเธอคิดว่าอีวาเป็นคนเอาไป และหลังจากที่อแมนด้าเดินชนกับอีวาที่ทางเดิน อแมนด้า แอน เจนนิเฟอร์ และอีวาก็ได้โต้เถียงกัน และลงท้ายด้วยการที่เจนนิเฟอร์ผลักแอนที่ยืนขวางอยู่ให้หลบไป แอนจึงขู่ไว้ว่าอย่ามาแตะต้องเธออีกไม่งั้นต้องเจอดีแน่ แต่อย่างไรก็ตามอแมนด้าก็หาคริสตัลของเธอพบในภายหลังซึ่งปรากฏว่าเธอเป็นคนไปเก็บไว้เองดังนั้นอแมนด้าจึงไปขอโทษและปรับความเข้าใจกับอีวาที่ด่วนตัดสินเธอสัปดาห์นี้ผู้เข้าแข่งขันต้องถ่ายรูปหมู่เพื่อโฆษณาให้กับกางเกงยีนส์ลีและต้องเปลือยหน้าอก ในระหว่างตัดสินสาวๆ ต้องทดสอบการเดินแบบในชุดสีชมพูและใส่รองเท้าที่เล็กกว่าปกติ 2 ไซส์ เจนนิเฟอร์เปิดโชว์ก้นที่เขียนไว้ว่า ANTM ระหว่างเดินไปหาผู้ตัดสิน คริสตี้และเจนนิเฟอร์ต้องกลายเป็นสองคนสุดท้าย และถึงแม้ผลงานที่ผ่านมาจะแข็งแกร่งกว่าเจนนิเฟอร์ แต่คริสตี้ก็ถูกคัดออกเนื่องจากขาดความมั่นใจจนแสดงออกมาในภาพผู้ที่ถูกเรียกชื่อคนแรก : แคสซี่ กริสแฮมผู้ที่ยืนเป็นสองคนสุดท้าย : เจนนิเฟอร์ ฟรอสต์ และ คริสตี้ กรอมเม็ทผู้ที่ถูกคัดออก : คริสตี้ กรอมเม็ทช่างถ่ายภาพ : ฮวงวาน รูแขกรับเชิญพิเศษ : เจ.อเล็กซานเดอร์, โยอันนา เฮาว์ส, แดนิโล, ริชชี่ ริช, เฮทเทอเรทสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ยาย่า ดาคอสต้าออกอากาศ : 20 ตุลาคม พ.ศ. 2547ผู้เข้าแข่งขันถูกส่งให้ไปเรียนการโพสท่ากับ เจนิส ดิคคินสัน และจากนั้นสาวๆ ต้องแข่งโพสท่าในชุดชั้นในที่ตู้กระจกหน้าร้าน ลา เพอร์ล่า  เคลลี่ได้รับคำชมว่าโพสได้เป็นธรรมชาติที่สุดและเธอก็เป็นผู้ชนะในการแข่งขันโดยรางวัลของเธอก็คือชุดชั้นในของในร้านมูลค่า 5,000 เหรียญ และเธอเลือกที่จะแบ่งรางวัลให้กับอแมนด้า และทอคคาราด้วยการถ่ายภาพในสัปดาห์นี้ผู้เข้าแข่งขันต้องโฆษณากระเป๋าถือดูนี่แอนด์โบรค โดยที่สาวๆ ต้องใส่รองเท้าสเก็ตด้วย มิสเตอร์เจย์ชื่นชอบในท่าโพสท์ของอแมนด้า นิโคล อีวา และยาย่ามาก ในขณะที่ภาพของเจนนิเฟอร์ แอน และเคลลี่ ไม่สามารถสร้างความประทับใจให้ได้เลยในห้องตัดสิน ภาพของเคลลี่ดูไร้อารมณ์ และกรรมการรู้สึกได้ถึงความกังวลและไม่มีพลังในรูปของเจนนิเฟอร์ ซึ่งทำให้พวกเธอต้องกลายเป็นสองคนสุดท้ายอีกครั้ง สุดท้ายแล้วคนที่ต้องกลับบ้านก็คือเจนนิเฟอร์เพราะกรรมการรู้สึกว่าเธอดูหมดไฟและไม่ได้ต้องการสิ่งนี้มากเท่าที่ควรผู้ที่ถูกเรียกชื่อคนแรก : อแมนด้า สวาฟฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : เจนนิเฟอร์ ฟรอสต์ และ เคลลี่ เจค็อบผู้ที่ถูกคัดออก : เจนนิเฟอร์ ฟรอสต์ช่างถ่ายภาพ : แมททิว จอร์แดน สมิทแขกรับเชิญพิเศษ : ไซมอน ดูแนน, เซฟยาส, เลนนี่ วาร์ดสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ทอคคาร่า โจนส์ออกอากาศ : 27 ตุลาคม พ.ศ. 2547อีวาและแอนรู้สึกรำคาญที่แคสซี่ไม่ยอมล้างจานของตัวเอง และยังวางทิ้งไว้ทั่วบ้าน แอนจึงใช้มีดเขียนบนบราวนี่ของแคสซี่ว่า “ล้างจานบราวนี่โง่ๆ ของเธอด้วย” ซึ่งทำให้แคสซี่ไม่พอใจมาก ในตอนอาหารค่ำยาย่าใส่เสื้อที่เขียนว่า "RESPEITO" ซึ่งเป็นภาษาโปรตุเกสแปลว่าการเคารพซึ่งกันและกัน โดยเธอบอกว่าเธอรู้สึกว่าทุกคนในบ้านขาดสิ่งนี้มากเธอก็เลยอยากใส่มันมาในคืนนี้ ทำให้แคสซี่รู้ว่าแอนเป็นคนเขียนบราวนี่ของเธอและมีปากเสียงกันขึ้น  ไทร่าได้พา เคท ดิลลอน เพื่อนนางแบบไซส์ใหญ่ของเธอซึ่งเคยเป็นโรค อโนเร็กเซีย ให้เข้ามาคุยกับสาวๆ ตัวต่อตัวเรื่องของสุขภาพและการนับถือตัวเอง แคสซี่เล่าว่าเวลาเธอกินมากเกินไปเธอจะเข้าไปล้วงคอตัวเองในห้องน้ำ และเธอปฏิเสธคำแนะนำของไทร่าเพราะไม่อยากยอมรับว่าตัวเองเป็นบูลิเมียในการแข่งขันสาวๆ ต้องวิ่งขึ้นบันได 14 ชั้น และถ่ายรูปทันที ยาย่าที่ขึ้นมาถึงเป็นคนแรกเป็นผู้ชนะและเธอเลือกทอคคาร่าไปสปาซึ่งเป็นรางวัลกับเธอด้วยในสัปดาห์นี้สาวๆ ต้องถ่ายรูปขณะกระโดดบนแทรมโบลินเพื่อโฆษณาให้กบเครื่องดื่มให้พลังงาน วายเจ สดิงเจอร์  ภาพของอแมนด้า อีวา และนิโคล ได้รับคำชมเปนอย่างมากในขณะที่ภาพของเคลลี่ แอน แคสซี่ และยาย่าดับสนิทเคลลี่และแอนตกเป็นสองคนสุดท้ายเพราะภาพที่ออกมาไร้ชีวิตชีวา แต่กรรมการเชื่อว่าแอนยังคงมีพลังและศักยภาพมากกว่าซึ่งนั่นทำให้เคลลี่ต้องถูกส่งกลับบ้านผู้ที่ถูกเรียกชื่อคนแรก : อีวา พิกฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : แอน มาร์คลีย์ และ เคลลี่ เจค็อบผู้ที่ถูกคัดออก : เคลลี่ เจค็อบช่างถ่ายภาพ : ซาช่า พีเฟล็กกิงแขกรับเชิญพิเศษ : จอย บิวเออร์, กัปตัน ลีฟ เวด, เคท ดิลลอนสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ทอคคาร่า โจนส์ออกอากาศ : 3 พฤศจิกายน พ.ศ. 2547ผู้เข้าแข่งขันต้องออกไปพบกับดีไซน์เนอร์ต่างๆ ท่าเดินของแคสซี่และโนเรลล์เรียกได้ว่าหายนะ บรรดาดีไซน์เนอร์ต่างประทับใจในตัวยาย่า และนั่นทำให้เธอเป็นผู้ชนะในการแข่งขัน ซึ่งรางวัลของเธอก็คือเสื้อผ้าต่างๆ จากบรรดาดีไซเนอร์ที่ออกไปพบ และเธอเลือกให้นิโคลได้แบ่งรางวัลกับเธอครึ่งนึงด้วยในสัปดาห์นี้สาวๆ ต้องถ่ายรูปโฆษณาให้กับรถยนต์ ฟอร์ด มัสแตง  โดยผู้เข้าแข่งขันต้องถ่ายรูปสองครั้งเพื่อแสดงถึงแต่ละบุคลิก 2 บุคลิก  ในห้องผู้ตัดสิน ทอคคาร่ากลายเป็นสองคนสุดท้าย เพราะรูปร่างของเธอดูจะค่อนข้างเป็นปัญหาในวงการนางแบบ แต่เป็นแคสซี่ที่ต้องตกรอบเพราะกรรมการรู้สึกว่าเธอดูหมดไฟและขาดพลังในการถ่ายรูปผู้ที่ถูกเรียกชื่อคนแรก : อีวา พิกฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : แคสซี่ กริสแฮม และ ทอคคาร่า โจนส์ผู้ที่ถูกคัดออก : แคสซี่ กริสแฮมช่างถ่ายภาพ : เดวิดแขกรับเชิญพิเศษ : นิโคล มิลเลอร์, ซินเทีย โรวลีย์, ไดแอน วอน ฟิวสเตนเบิร์ก, นาเนตตา เลพอเร่, เจ.อเล็กซานเดอร์, มาร์ค โบวเวอร์สาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ทอคคาร่า โจนส์ออกอากาศ : 10 พฤศจิกายน พ.ศ. 2547ผู้เข้าแข่งขันได้รับคำแนะนำจากสไตล์ลิสต์ รีเบคก้า ไวน์เบิร์ก และถูกพาไปที่ร้าน Century 21 เพื่อเปลี่ยนเครื่องแต่งกายให้พร้อมที่จะเดินบนพรมแดง ยาย่าสร้างความประทับใจในการสัมภาษณ์ และเป็นผู้ชนะในการแข่งขัน โดยเธอได้เลือกโนเรลล์ ให้ได้ร่วมรับรางวัลกับเธอด้วยในการถ่ายภาพครั้งนี้ ต้องถ่ายแบบ บิวตี้ช็อต กับเครื่องเพชรของ เวอร์ราจิโอ้ อีวารู้สึกหวาดผวาเมื่อรู้ว่าต้องถ่ายภาพกับแมงมุมทารันทูล่าด้วย ซึ่งเธอแสดงความกลัวและร้องไห้ตลอดเวลาในการถ่าย  ในห้องตัดสิน รีเบคก้าตำหนิว่ายาย่าต้องการแสดงความเป็นแอฟริกันของเธอมากเกินไป กรรมการรู้สึกว่าภาพของอีวา นิโคล และยาย่าดูโดดเด่นกว่าทุกๆ คน ในขณะที่แอน ทอคคาร่า และโนเรลล์ตายสนิทในภาพ  และคนที่ถูกคัดออกคือทอคคาร่า เพราะกรรมการรู้สึกว่าเธอได้สูญเสียความเป็นตัวของตัวเองและความต้องการไปแล้วผู้ที่ถูกเรียกชื่อคนแรก : นิโคล โบรัดผู้ที่ยืนเป็นสองคนสุดท้าย : แอน มาร์คลีย์ และ ทอคคาร่า โจนส์ผู้ที่ถูกคัดออก : ทอคคาร่า โจนส์ช่างถ่ายภาพ : บิล ดิโอดาโต้แขกรับเชิญพิเศษ : พอล วิลมอท, แฮมพ์ตัน คาร์นีย์, ซาร่าห์ ไพเพอร์, รีเบคก้า ไวน์เบิร์กสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : ทอคคาร่า โจนส์ออกอากาศ : 17 พฤศจิกายน พ.ศ. 2547ผู้เข้าแข่งขันที่เหลือต้องมาเรียนการแสดงกับเทย์ ดิกส์  จากนั้นไทร่าได้ออกมาบอกกับสาวๆ ให้เก็บกระเป๋าเดินทางไปกรุงโตเกียว ประเทศญี่ปุ่น โดยยาย่าซึ่งเป็นผู้ชนะในการแข่งขันการแสดงได้รางวัลคือการขึ้นเครื่องชั้นเฟริสต์คลาส โดยเธอเลือกให้อแมนด้าไปกับเธอด้วย ในขณะที่สาวๆ ที่เหลือต้องไปนั่งชั้นประหยัดสำหรับครั้งนี้ผู้เข้าแข่งขัญต้องถ่ายโฆษณาซุปกระป๋องแคมพ์เบลเป็นภาษาญี่ปุ่น อีวาซึ่งถ่ายโฆษณาออกมาได้ไม่ดีแต่เธอกลับคิดว่าโนเรลล์อาจต้องตกรอบซึ่งทำให้แอนซึ่งสนิทกับโนเรลล์เหมือนกันไม่พอใจ ในห้องตัดสินสาวๆ จะต้องถ่ายโฆษณาลูกพลัมดองอุเมะโบชิ ยาย่าแสดงทัศนคติด้านลบออกมาเมื่อเธอคายสินค้าออกมาระหว่างถ่ายซึ่งทำให้เธอต้องตกเป็นสองคนสุดท้าย แต่เป็นนิโคลที่ต้องกลับบ้านเพราะกรรมการรู้สึกว่าเธอไม่เป็นที่น่าจดจำ ไม่มีอะไรโดดเด่นและไทร่าเองก็ยังลืมเธอในการตัดสินผู้ที่ถูกเรียกชื่อคนแรก : แอน มาร์คลีย์ผู้ที่ยืนเป็นสองคนสุดท้าย : นิโคล โบรัด และ ยาย่า ดาคอสต้าผู้ที่ถูกคัดออก : นิโคล โบรัดผู้กำกับโฆษณา : เซนเต้ โยชิอิแขกรับเชิญพิเศษ : เทย์ ดิกส์, แซนดี้ ชูริน, โทโกะ นิวะสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : อีวา พิกฟอร์ดออกอากาศ : 24 พฤศจิกายน พ.ศ. 2547ประมวลเหตุการณ์ทั้งหมดที่ผ่านมาของผู้เข้าแข่งขันจาก 14 คน เหลือ 5 คนสุดท้าย และภาพเหตุการณ์ต่างๆ ที่ไม่เคยเปิดเผยมาก่อนออกอากาศ : 1 ธันวาคม พ.ศ. 2547การแข่งขันในครั้งนี้สาวๆ ต้องเรียนรู้และแสดงพิธีการชงชาแบบญี่ปุ่นที่ถูกต้อง ยาย่าเป็นผู้ชนะในการแข่งโดยรางวัลของเธอก็คือการไปแช่บ่อน้ำพุร้อน และเธอเลือกให้อแมนด้าไปกับเธอด้วย เพราะความขัดแย้งก่อนหน้านี้ทำให้อีวามองหน้าแอนกับโนเรลล์ไม่ติด แต่ภายหลังอีวาได้มาปรับความเข้าใจกับโนเรลล์และบอกว่ายังรักและยังอยากจะเป็นเพื่อนกับเธออยู่เหมือนเดิม ในสับดาห์นี้ผู้เข้าแข่งขันต้องถ่ายรูปโฆษณาให้กับโทรศัพท์มือถือ ที-โมบาย สาวๆ ต้องแต่งตัวเป็นเกอิชา โดยมีไทร่ามาโพสให้ดูเป็นตัวอย่างก่อน ภาพของอีวา อแมนด้า และโนเรลล์ดูโดดเด่น ในขณะที่ภาพของแอนกับยาย่าไม่สามารถที่จะสร้างความประทับใจได้  ไทร่ารู้สึกผิดหวังกับท่าทีของอีวา และโนเรลล์เป็นผู้ที่ต้องตกรอบเพราะบุคลิกและท่าเดินอันแสนจะเลวร้ายในห้องตัดสินผู้ที่ถูกเรียกชื่อคนแรก : อแมนด้า สวาฟฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : แอน มาร์คลีย์ และ โนเรลล์ แวน เฮิร์คผู้ที่ถูกคัดออก : โนเรลล์ แวน เฮิร์คช่างถ่ายภาพ : เอมิโกะแขกรับเชิญพิเศษ : โทโกะ นิวะ, คุนิเนโอะ โคอิเคะ, โยโกะ ทาคาฮาชิ, เคียวโกะ ฮิกะสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : อีวา พิกฟอร์ดออกอากาศ : 8 ธันวาคม พ.ศ. 2547ในการแข่งขันสัปดาห์นี้ผู้เข้าแข่งขันต้องสร้างสไตล์แฟชั่นบนถนนของโตเกียวให้กับตัวเอง แอนกับอีวาจับคู่ไปด้วยกัน ยาย่ามาสายในการแข่งขันทำให้อีวากลายเป็นผู้ชนะไป และเธอเลือกให้แอนมาแบ่งรางวัลร่วมกับเธอด้วย ซึ่งก็คือไปทานอาหารเช้ากับนักออกแบบเครื่องประดับ มิคิโมโตะ และทั้งคู่ก็ยังได้รับสร้อยมุกเป็นของขวัญอีกด้วยการถ่ายรูปครั้งนี้สาวๆ ต้องขี่มอเตอร์ไซด์และถ่ายรูปออกมาในแนวการ์ตูน  ภาพของอแมนด้ากับยาย่าได้รับคำชมจากกรรมการอย่างมาก ในขณะที่ภาพของแอนกับอีวาดูไม่มีพลัง และแอนก็ต้องตกรอบเพราะกรรมการรู้สึกว่าเธอขาดความกระตือรือล้นและความรับผิดชอบในตัวเอง และหลังจากถูกคัดออกแอนได้เข้าไปกอดลาทุกคนยกเว้นอีวา ซึ่งนั่นอาจหมายถึงจุดสิ้นสุดมิตรภาพของพวกเธอผู้ที่ถูกเรียกชื่อคนแรก : อแมนด้า สวาฟฟอร์ดผู้ที่ยืนเป็นสองคนสุดท้าย : แอน มาร์คลีย์ และ อีวา พิกฟอร์ดผู้ที่ถูกคัดออก : แอน มาร์คลีย์ช่างถ่ายภาพ : ทาเคชิ มิเอซากิแขกรับเชิญพิเศษ : เจ.อเล็กซานเดอร์, โทโยฮิโกะ มิคิโมโตะ, ซาวะ วอชเตอร์, แคโรไลน์ ลอนดอน, ชินโง, ฮิโตมิสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : อีวา พิกฟอร์ดออกอากาศ : 15 ธันวาคม พ.ศ. 2547ในรอบสุดท้ายของฤดูกาลที่ 3 อแมนด้า  ยาย่าและอีวา ต้องไปถ่ายรูปลงนิตยสารให้กับคัฟเวอร์เกิร์ล  ในห้องตัดสินภาพของผู้แข่งขันทั้งสามได้รับคำชมเป็นอย่างมากซึ่งนั่นทำให้คณะกรรมการต้องคิดหนักในการคัดออก ยาย่าถูกเรียกชื่อเป็นคนแรก และอแมนด้ากับอีวาต้องกลายเป็นสองคนสุดท้าย และด้วยบุคลิกที่ดูเป็นมิตรและเข้าถึงง่ายของอีวาทำให้เธอได้เข้าสู่รอบสองคนสุดท้าย และนั่นทำให้อแมนด้าต้องใจสลายผู้ที่ถูกเรียกชื่อคนแรก : ยาย่า ดาคอสต้าผู้ที่ยืนเป็นสองคนสุดท้าย : อแมนด้า สวาฟฟอร์ด และ อีวา พิกฟอร์ดผู้ที่ถูกคัดออก : อแมนด้า สวาฟฟอร์ดช่างถ่ายภาพ : ชู เซนสึในการตัดสินครั้งสุดท้าย ยาย่าและอีวาต้องเดินบนรันเวย์แบบเซน โดยการไม่แสดงสีหน้าขณะเดิน กรรมการลงความเห็นว่าถึงการเดินและผลงานที่ผ่านมาของยาย่าดีมาก แต่ทัศนคติของเธอที่ดูหยิ่งจนเกินไป ในขณะที่อีวาดูเป็นมิตรและเข้าถึงง่ายกว่า และนั่นคือสิ่งที่ทำให้อีวาเป็นผู้ชนะของฤดูกาลที่ 3ผู้ชิงตำแหน่งสองคนสุดท้าย : อีวา พิกฟอร์ด และ ยาย่า ดาคอสต้าอเมริกา เน็กซต์ ทอป โมเดล : อีวา พิกฟอร์ดแขกรับเชิญพิเศษ : เจ.อเล็กซานเดอร์, โนริโกะ ฟุคุชิมะสาวคัฟเวอร์เกิร์ลประจำสัปดาห์ : อีวา พิกฟอร์ด     ผู้ชนะการแข่งประจำสัปดาห์     ผู้ชนะผลโหวดสาวคัฟเวอร์เกิร์ลประจำสัปดาห์     ผู้ชนะผลโหวดสาวคัฟเวอร์เกิร์ล และ การแข่งประจำสัปดาห์     ผู้ชนะผลโหวดสาวคัฟเวอร์เกิร์ล และ ออกจากการแข่งขัน     ผู้ที่ถูกคัดออกนอกห้องตัดสิน     ผู้ที่ถูกคัดออกจากการแข่งขัน     ผู้ที่ชนะเลิศการแข่งขันตอนที่ 10 เป็นตอนประมวลภาพซ้ำในการแข่งขันอแมนด้า : ย้อมสีไอซ์บลอนด์ และต่อให้ยาวถึงกลางหลังแอน : ตัดให้สั้นระดับคาง และ ทำไฮไลท์สีบลอนด์แคสซี่ : ต่อผมยาว สีบลอนด์ ให้หนา ต่อผมยาวสีดำ ทำทรงแอฟโฟร์ และพาไปรักษาสิวสัปดาห์ที่ 2 : ถ่ายภาพชุดว่ายน้ำ ที่ชายหาดจาไมกาสัปดาห์ที่ 3 : ถ่ายภาพบิวตี้ ช็อท ให้กับ นีเวีย โลชั่นทามือสัปดาห์ที่ 4 : ถ่ายภาพเปลือยท่อนบน ให้กับ
