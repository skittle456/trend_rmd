โครงข่ายระบบรถไฟฟ้าในประเทศไทย เป็นระบบขนส่งมวลชนแบบรางในประเทศไทย เริ่มเปิดให้บริการเป็นครั้งแรกเมื่อปี พ.ศ. 2542 โดยกรุงเทพมหานครเป็นเมืองแรกที่มีระบบรถไฟฟ้า ตามมาด้วยนนทบุรี และสมุทรปราการ ตามลำดับรถไฟฟ้าในกรุงเทพมหานครและปริมณฑล เป็นระบบขนส่งมวลชนแบบรางในกรุงเทพมหานครและปริมณฑล เริ่มเปิดให้บริการเป็นครั้งแรกเมื่อปี พ.ศ. 2542 ในเส้นทางหมอชิต - อ่อนนุชและสนามกีฬาแห่งชาติ - สะพานตากสิน โดยในปัจจุบันมีรถไฟฟ้าที่เปิดให้บริการแล้วทั้งสิ้น 5 สาย ครอบคลุมระยะทาง 110.29 กิโลเมตรกรุงเทพมหานครเตรียมผุดระบบขนส่งมวลชนรอง รถราง  รอบพื้นที่เกาะรัตนโกสินทร์เพื่อแก้ไขปัญหาจราจรและส่งเสริมการท่องเที่ยวในเขตเมืองเก่า พร้อมเชื่อมต่อโครงข่ายรถไฟฟ้าสายสีน้ำเงินและสายสีส้มในอนาคตรถไฟฟ้าในเทศบาลนครขอนแก่น เป็นโครงการระบบขนส่งมวลชนแบบรางเบาในพื้นที่ 5 เทศบาล ประกอบด้วย เทศบาลนครขอนแก่น เทศบาลเมืองศิลา เทศบาลตำบลเมืองเก่า เทศบาลตำบลสำราญ และ เทศบาลตำบลท่าพระโครงการนี้ได้รับการอนุมัติในหลักการแล้ว ความคืบหน้าตุลาคม 2560 บริษัท ขอนแก่น ทรานซิท ซิสเต็ม จำกัด  เจ้าของโครงการได้มีการประกวดราคาและให้ยื่นข้อเสนอโครงการก่อสร้าง และเดือนกุมภาพันธ์ 2561 อยู่ในระหว่างการเตรียมประกาศรายชื่อผู้ผ่านการคัดเลือกในการก่อสร้างโครงการ ล่าสุด  ที่ประชุม สนข. มีมติเห็นชอบรายงานการวิเคราะห์ผลกระทบสิ่งแวดล้อม  โดยหลังจากนี้ไป สนข. จะนำเรื่องเข้าที่ประชุมคณะกรรมการจัดระบบการจราจรทางบก  และเสนอต่อคณะรัฐมนตรี  ต่อไปตามลำดับ23 เมษายน พ.ศ. 2561 การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย ได้ประกาศเปิดรับฟังความคิดเห็นต่อร่างพระราชกฤษฎีกาให้การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย ดำเนินกิจการรถไฟฟ้าในจังหวัดขอนแก่น โดยจะรวบรวมความคิดเห็นจากทุกฝ่าย ซึ่งสิ้นสุดวันที่ 25 พฤษภาคม 2561   และมีมติอย่างเป็นทางการจากกระทรวงคมนาคมให้เทศบาลขอนแก่นดำเนินการก่อสร้างรถไฟฟ้าด้วยตนเอง โดยให้ บริษัท ขอนแก่น ทรานซิท ซิสเท็ม จำกัด หรือ เคเคทีเอส เป็นผู้ดำเนินการรถไฟฟ้าสายแรกของขอนแก่น อย่างไรก็ตามเทศบาลฯ มีความยินดีที่จะให้เคเคทีเอสเข้าร่วมดำเนินการหาก การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย มีความต้องการที่จะก่อสร้างรถไฟฟ้าเส้นทางอื่น ๆ ให้ครบตามแผน28 สิงหาคม พ.ศ. 2561 การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย เผยแพร่สรุปผลการรับฟังความคิดเห็นต่อร่างพระราชกฤษฎีกาให้การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย ดำเนินกิจการรถไฟฟ้าในจังหวัดขอนแก่น โดยทาง รฟม. จะชะลอการดำเนินการ และให้การสนับสนุนองค์กรปกครองส่วนท้องถิ่นในการดำเนินการโครงการนี้ต่อไป17 ตุลาคม พ.ศ. 2561 ที่ประชุมคณะกรรมการจัดระบบการจราจรทางบก  มีมติเห็นชอบผลการศึกษาออกแบบรายละเอียดระบบขนส่งสาธารณะในเขตจังหวัดขอนแก่นและผลกระทบสิ่งแวดล้อมตามที่ สำนักงานนโยบายและแผนการขนส่งและจราจร ได้ศึกษาไว้ และอนุญาตให้จังหวัดขอนแก่นเป็นผู้พัฒนาและบริหารจัดการโครงการตามแผนพัฒนาระบบขนส่งสาธารณะเมืองขอนแก่นดังกล่าว เฉพาะในเส้นทางนำร่องสายสีแดง   โดยโครงการจะเป็นระบบ รถไฟฟ้ารางเบา  มีจำนวนสถานีทั้งหมด 16 สถานี เป็นโครงสร้างยกระดับ 6 สถานี และระดับดิน 10 สถานี ระยะทาง 22.8 กิโลเมตร คาดว่าจะสามารถวางศิลาฤกษ์ได้ปลายปี พ.ศ. 2561รถไฟฟ้าในเทศบาลนครเชียงใหม่ เป็นโครงการระบบขนส่งมวลชนแบบรางในเทศบาลนครเชียงใหม่ เนื่องจากเชียงใหม่เป็นเมืองที่มีนักท่องเที่ยวมาก ทำให้เทศบาลนครเชียงใหม่ต้องพัฒนาระบบขนส่งมวลชนเพื่อรองรับนักท่องเที่ยว โดยเสนอโครงการระบบรถไฟฟ้ารางเดี่ยวหรือโมโนเรล ทั้งหมด 4 สาย ซึ่งโครงการนี้ต้องใช้เงินกว่า 8 ล้านบาท และใช้เวลาก่อสร้างประมาณ 5 ปี ปัจจุบัน โครงการนี้ยังอยู่ในขั้นตอนออกแบบและศึกษาโครงการเทศบาลนครนครราชสีมา เป็น 1 ใน 3 เทศบาล ที่มีโครงการก่อสร้างรถราง โดยเสนอโครงการระบบรถรางสำนักงานนโยบายและแผนการขนส่งและจราจร  ได้ตั้งงบประมาณเพื่อศึกษาความเป็นไปได้ของขนส่งมวลชนในเมืองนครราชสีมา โดยให้มหาวิทยาลัยเทคโนโลยีสุรนารี  เป็นผู้ศึกษาและออกแบบ  รูปแบบขนส่งที่เลือกใช้คือ ระบบรถรางเบาระดับพื้น ความคืบหน้าวันที่ 20 ตุลาคม 2560  ได้มีการประชุมฟังความคิดเห็นของประชาชนในพื้นที่เป็นครั้งสุดท้ายเพื่อจัดทำเอกสารส่งคณะรัฐมนตรีพิจารณา23 เมษายน 2561 การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย ได้ประกาศเปิดรับฟังความคิดเห็นต่อร่างพระราชกฤษฎีกาให้การรถไฟฟ้าขนส่งมวลชนแห่งประเทศไทย ดำเนินกิจการรถไฟฟ้าในจังหวัดนครราชสีมา โดยจะรวบรวมความคิดเห็นจากทุกฝ่าย ตั้งแต่วันที่ 23 เมษายน 2561 ถึงวันที่ 25 พฤษภาคม 2561 ปัจจุบันได้ผ่านความเห็นชอบจาก สำนักงานนโยบายและแผนการขนส่งและจราจร และกระทรวงคมนาคมแล้ว  จะเสนอคณะกรรมการจัดระบบการจราจรทางบก  พิจารณาต่อไป เทศบาลนครพิษณุโลก เป็น 1 ใน 3 เทศบาล ที่มีโครงการก่อสร้างรถราง โดยเสนอโครงการระบบรถรางเทศบาลนครอุดรธานี เป็นโครงการก่อสร้างรถราง โดยเสนอโครงการระบบรถไฟฟ้ารางเดี่ยวหรือโมโนเรลวันที่ 30 ตุลาคม 2561 องค์การบริหารส่วนจังหวัดสงขลา จัดการประชุมเพื่อประเมินความสนใจของภาคเอกชน  ใช้ระบบการลงทุนกับภาคเอกชน  เป็นระยะเวลา 30 ปี  ประกอบด้วย 12 สถานี แนวเส้นทางเริ่มต้นจากสถานีคลองหวะ ถึงสถานีรถตู้ ระยะทาง 12.54 กิโลเมตร สามารถเริ่มก่อสร้างได้ปี พ.ศ. 2563 โดยเป็นโครงการระบบรถไฟฟ้ารางเดี่ยวหรือโมโนเรลรถไฟฟ้าในเทศบาลนครภูเก็ต เป็นเป็นโครงการระบบขนส่งมวลชนแบบรางในเทศบาลนครภูเก็ต เนื่องจากภูเก็ตเป็นเมืองที่มีนักท่องเที่ยวมาก ทำให้เทศบาลนครภูเก็ตต้องพัฒนาระบบขนส่งมวลชนเพื่อรองรับนักท่องเที่ยว โดยเสนอโครงการระบบรถไฟฟ้ารางเบาและรถราง ปัจจุบันอยู่ระหว่างให้ที่ปรึกษาทำการศึกษารูปแบบการเปิดให้เอกชนร่วมทุนตามขั้นตอนพ.ร.บ.จัดซื้อจัดจ้างโดยอยู่ระหว่างการศึกษาและออกแบบโครงการเพื่อเตรียมเสนอคมนาคมพิจารณา คาดว่าจะใช้เวลาก่อสร้างประมาณ 3 ปีเทศบาลนครภูเก็ต เป็น 1 ใน 3 เทศบาล ที่มีโครงการก่อสร้างรถรางรายชื่อสถานีรถไฟรางเบาสถานีท่านุ่นสถานีท่าฉัตรไชยสถานีประตูเมืองภูเก็ตสถานีท่าอากาศยานภูเก็ตสถานีถลางสถานีอนุสาวรีย์ท้าวเทพกษัตรีย์-ท้าวศรีสุนทรสถานีเกาะแก้วสถานีขนส่งภูเก็ต2สถานีมหาวิทยาลัยราชภัฏภูเก็ตสถานนีทุ่งคาสถานีเมืองเก่าสถานีหอนาฬิกาสถานีบางเหนียวสถานีห้องสมุดประชาชนสถานีสะพานหินสถานีศักดิเดชน์สถานีดาวรุ่งสถานีวิชิตสถานีเจ้าฟ้าตะวันออกสถานีป่าหล่ายสถานีบ้านโคกโตนดสถานีฉลองรถไฟฟ้าในเมืองพัทยา เป็นเป็นโครงการระบบขนส่งมวลชนแบบรางในเมืองพัทยา เนื่องจากพัทยาเป็นเมืองที่มีนักท่องเที่ยวมาก ทำให้เมืองพัทยาต้องพัฒนาระบบขนส่งมวลชนเพื่อรองรับนักท่องเที่ยว โดยเสนอโครงการระบบรถไฟฟ้ารางเดี่ยวหรือโมโนเรล ปัจจุบันโครงการนี้ อยู่ระหว่างพักการดำเนินการชั่วคราว เนื่องจากได้รับการคัดค้านจากประชาชนเป็นส่วนใหญ่รายชื่อสถานีรถไฟฟ้า:สถานีแหลมบาลีไฮสถานีพัทยาใต้สถานีพัทยากลางสถานีพัทยาเหนือสถานีศาลาว่าการเมืองพัทยาสถานีขนส่งพัทยาสถานีโพธิสารสถานีสุขุมวิท-พัทยา 33สถานีชัยพรวิถีสถานีขนส่งพัทยาใหม่รถไฟฟ้าในกรุงเทพมหานครและปริมณฑล
